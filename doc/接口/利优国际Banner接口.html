	0003  保存新建Banner接口
	
	  请求URI：http://192.168.1.19:10003/manage/banner/saveBanner
	  请求报文(用户登录后才能查看，后台会拿当前登录用户ID):
		"param"：{
			"category":"index",			//栏目分类
			"area":"湖南长沙",			//所属地区
			"status":"1",				//开启状态
			"startDate":"2018-9-15",	//开始时间
			"endDate":"2018-9-20",		//结束时间
			"murl":"/media?id=1001",	//媒介url，图片或视频url
			"aurl":"/action?id=1002"	//链接url，点击后跳转url
			}
	  返回报文：
	    {
			"pageNumber":0,
			"pageSize":10,
			"totalCount":1,
			"list":[
				{
					"path":"test",
					"createTime":"2018-08-27 19:12:25",
					"count":1,
					"updateTime":"2018-08-27 19:12:25",
					"id":1,
					"img":"xxx" //图片地址
					"state":1,
					"title":"dd",
					"userId":1,
					"userName":"ss", //对应的用户昵称
					"userPic":"xx"  //用户头像地址
				}
			]
		}
	

	0004  保存修改Banner接口
	
	  请求URI：http://192.168.1.19:10003/manage/banner/updateBanner
	  请求报文:
		"param"：{
			"id":"25",					//
			"category":"index",			//栏目分类
			"area":"湖南长沙",			//所属地区
			"status":"0",				//开启状态
			"startDate":"2018-9-15",	//开始时间
			"endDate":"2018-9-20",		//结束时间
			"murl":"/media?id=1001",	//媒介url，图片或视频url
			"aurl":"/action?id=1002"	//链接url，点击后跳转url
		}
	  返回报文：
	    {
			"message": "更新成功",
			"result": 0
		}
	

	0005  删除Banner接口
	
	  请求URI：http://192.168.1.19:10003/manage/banner/deleteBanner
	  请求报文:

			"id":"25"	//Banner id
			}
	  返回报文：
	    {
			"message": "删除成功",
    		"result": 0
		}
