    0002  添加视频接口
	
	  请求URI：http://192.168.1.200:10002/video/manage/upload
	  请求报文(用户登录后才能上传,用户ID为后台获取):
		"param"：{   
			"title":"xxx",	//标题
			"path":"",    //上传视频路径
			"times":"09:00-21:00" //视频时长
			"img":"xxx" //图片地址
			}
	  返回报文：
	    {
			"data":{
				"data":"{}",
				"message":"保存成功",
				"result":0
			},
			"message":"ok",
			"result":0
		}
	
	
	0003  分页查询我的视频接口
	
	  请求URI：http://192.168.1.200:10002/video/manage/queryPageByMyVideo
	  请求报文(用户登录后才能查看，后台会拿当前登录用户ID):
		"param"：{   
			"pageNumber":0,	//当前页
			"pageSize":10,    //页面大小
			}
	  返回报文：
	    {
			"pageNumber":0,
			"pageSize":10,
			"totalCount":1,
			"list":[
				{
					"path":"test",
					"createTime":"2018-08-27 19:12:25",
					"count":1,
					"updateTime":"2018-08-27 19:12:25",
					"id":1,
					"img":"xxx" //图片地址
					"state":1,
					"title":"dd",
					"userId":1,
					"userName":"ss", //对应的用户昵称
					"userPic":"xx"  //用户头像地址
				}
			]
		}

	0006  用户为视频点赞接口
		
		  请求URI：http://192.168.1.200:10002/video/manage/videoThumb
		  请求报文(type:1--视频点赞，2--评论点赞 ,userId:手机端唯一编号):
			"param"：{   
				"userId":ddd,	//手机端唯一编号
				"type":1 //1--视频点赞，2--评论点赞
				"aId":aaa //视频ID或评论ID，与上面的type对应
				}
		  返回报文：
			{
				"data":{
					"data":"{}",
					"message":"修改成功",
					"result":0
				},
				"message":"ok",
				"result":0
			}

	0009  查询当前视频的评论接口

	  请求报文

			"pageNumber":1,	
			"pageSize":10,
			"id":1 //视频ID,
			"deviceId":"xx"  查询点赞专用,手机的唯一编号

	    {
			"data": {
				"list": [
					{
						"commentThumb": 0, //点赞的总数
						"content": "test", //标题
						"createTime": "2018-08-27 20:09:10",
						"id": 3,
						"isThumb": false,  //是否点赞：false 没有，true:已点
						"iscomment": 1,   //
						"state": 1,
						"updateTime": "2018-08-27 20:08:58",
						"userId": 1,
						"videoId": 1，
						"userName":"ss", //对应的用户昵称
					    "userPic":"xx"  //用户头像地址
					}
				],
				"pageNumber": 1,
				"pageSize": 10,
				"totalCount": 1,
				"totalPage": 1
			},
			"message": "OK",
			"result": 0
		}
	
	
	0010  添加当前视频的评论接口

			}

	    {
			"data":{
				"data":"{}",
				"message":"保存成功",
				"result":0
			},
			"message":"ok",
			"result":0
		}
